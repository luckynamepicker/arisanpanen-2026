        const names=["Supe","uki","nadia","alif&alifa","Risda","elaa","nur auliya","aisyah","husnah","muliati.s","habib"];

        const result=document.getElementById('result');

        const colors=['#ff6b6b','#feca57','#48dbfb','#ff9ff3','#54a0ff','#5f27cd','#00d2d3','#ff9f43','#00b894','#e17055','#fd79a8','#6c5ce7'];

        const total=names.length;

        const angle=360/total;

        let spinning=false;


        // Create particles

        for(let i=0;i<40;i++){

            const p=document.createElement('div');

            p.className='particle';

            p.style.animationDelay=Math.random()*6+'s';

            document.getElementById('particles').appendChild(p);

        }


        // Create wheel

        function createWheel(){

            wheel.innerHTML='';

            names.forEach((name,i)=>{

                const seg=document.createElement('div');

                seg.className='segment'+(i===0?' supe-segment':'');

                seg.style.background=`linear-gradient(135deg,${colors[i%colors.length]},${colors[(i+3)%colors.length]})`;

                seg.style.transform=`rotate(${i*angle}deg)`;

                seg.innerHTML=`<div>${name}</div>`;

                wheel.appendChild(seg);

            });

        }


        // Sound - Spin

        function playSpinSound(){

            try{

                const ctx=new(window.AudioContext||window.webkitAudioContext)();

                const osc=ctx.createOscillator();

                const gain=ctx.createGain();

                osc.connect(gain);

                gain.connect(ctx.destination);

                osc.frequency.setValueAtTime(600,ctx.currentTime);

                osc.frequency.exponentialRampToValueAtTime(80,ctx.currentTime+4);

                gain.gain.setValueAtTime(0.2,ctx.currentTime);

                gain.gain.exponentialRampToValueAtTime(0.01,ctx.currentTime+4);

                osc.start();

                osc.stop(ctx.currentTime+4);

            }catch(e){}

        }


        // Sound - Win

        function playWinSound(){

            try{

                const ctx=new(window.AudioContext||window.webkitAudioContext)();

                [523,659,784,1047].forEach((freq,i)=>{

                    setTimeout(()=>{

                        const osc=ctx.createOscillator();

                        const gain=ctx.createGain();

                        osc.connect(gain);

                        gain.connect(ctx.destination);

                        osc.frequency.value=freq;

                        osc.type='sine';

                        gain.gain.setValueAtTime(0.3,ctx.currentTime);

                        gain.gain.exponentialRampToValueAtTime(0.01,ctx.currentTime+0.4);

                        osc.start();

                        osc.stop(ctx.currentTime+0.4);

                    },i*150);

                });

            }catch(e){}

        }


        // Confetti

        function launchConfetti(){

            const colors=['#feca57','#ff6b6b','#48dbfb','#ff9ff3','#54a0ff','#00b894'];

            for(let i=0;i<80;i++){

                const c=document.createElement('div');

                c.className='confetti';

                c.style.left=Math.random()*100+'%';

                c.style.background=colors[Math.floor(Math.random()*colors.length)];

                c.style.animationDuration=(Math.random()*2+2)+'s';

                c.style.animationDelay=Math.random()*0.5+'s';

                document.body.appendChild(c);

                setTimeout(()=>c.remove(),4000);

            }

        }


        // Spin function - SUPE ALWAYS WINS!

        function spin(){

            if(spinning)return;

            spinning=true;

            spinBtn.disabled=true;

            spinBtn.textContent='🎰 MEMUTAR...';

            result.textContent='⏳ Sedang berputar...';

            result.style.animation='none';

            

            playSpinSound();

            

            // Random spins + land on SUPE (index 0)

            const spins=5+Math.random()*5;

            const supeAngle=90+(angle/2); // Position for Supe

            const totalRotation=(spins*360)+supeAngle;

            

            wheel.style.transform=`rotate(-${totalRotation}deg)`;

            

            setTimeout(()=>{

                spinning=false;

                spinBtn.disabled=false;

                spinBtn.textContent='🔄 PUTAR LAGI';

                result.innerHTML='🏆 <span style="color:#feca57;font-size:1.3em">SUPE MENANG!</span> 🏆';

                playWinSound();

                launchConfetti();

            },4500);

        }
